<script>
    document.getElementById('plant-id-form').addEventListener('submit', async function(event) {
        event.preventDefault();

        const fileInput = document.getElementById('image-upload');
        const file = fileInput.files[0];

        if (!file) {
            alert('Please select an image first.');
            return;
        }

        const API_KEY = '2b10a24wwA5Ro9AlULI13FNze'; 
        const API_URL = 'https://my-api.plantnet.org/v2/identify/all?api-key=' + API_KEY;
        const resultsDiv = document.getElementById('results');

        resultsDiv.innerHTML = '<p>Identifying your plant... Please wait.</p>';

        const formData = new FormData();
        formData.append('images', file);
        formData.append('organs', 'leaf');

        try {
            const response = await fetch(API_URL, {
                method: 'POST',
                body: formData
            });

            if (!response.ok) {
                const errorData = await response.json();
                const errorMessage = (errorData.error && errorData.error.message) ? errorData.error.message : JSON.stringify(errorData, null, 2);
                throw new Error(errorMessage || 'An unknown error occurred.');
            }

            const data = await response.json();
            displayResults(data);

        } catch (error) {
            resultsDiv.innerHTML = '<p style=\'color: red;\'>Error: ' + error.message + '</p>';
            console.error('API Error:', error);
        }
    });

    function displayResults(data) {
        const resultsDiv = document.getElementById('results');
        resultsDiv.innerHTML = '';

        if (data.results && data.results.length > 0) {
            const bestMatch = data.results[0];
            const bestScore = (bestMatch.score * 100).toFixed(2);
            
            resultsDiv.innerHTML = '<p><strong>Common Name:</strong> ' + (bestMatch.species.commonNames[0] || 'N/A') + '</p>' +
                                   '<p><strong>Scientific Name:</strong> <em>' + bestMatch.species.scientificName + '</em></p>' +
                                   '<p><strong>Confidence:</strong> ' + bestScore + '%</p>' +
                                   '<p><strong>Family:</strong> ' + bestMatch.species.family.scientificName + '</p>' +
                                   '<p><em>Note: This is the most likely match.</em></p>';
        } else {
            resultsDiv.innerHTML = '<p>Sorry, no matching plant was found. Please try another picture.</p>';
        }
    }
</script>
